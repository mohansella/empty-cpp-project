cmake_minimum_required(VERSION 3.11)

project(MyApp)

#prepare conan
if(NOT EXISTS "${CMAKE_BINARY_DIR}/conan.cmake")
   message(STATUS "Downloading conan.cmake from https://github.com/conan-io/cmake-conan")
   file(DOWNLOAD "https://github.com/conan-io/cmake-conan/raw/v0.14/conan.cmake"
				 "${CMAKE_BINARY_DIR}/conan.cmake")
endif()

#resolve dependencies
include(${CMAKE_BINARY_DIR}/conan.cmake)
conan_cmake_run(CONANFILE conanfile.txt BASIC_SETUP CMAKE_TARGETS BUILD missing BUILD_TYPE ${CONAN_BUILD_TYPE})

#find locations
include(${CMAKE_BINARY_DIR}/conanbuildinfo.cmake)
conan_basic_setup()

#list cpp files
list(APPEND _cpp_files
    src/main.cpp
    #append more fils here
)

#add exe module with include folder and source files above
set(ExeModuleName MyApp)
add_executable(${ExeModuleName} ${_cpp_files})
target_include_directories(${ExeModuleName} PUBLIC include)

#link needed dependencies
target_link_libraries(${ExeModuleName} PUBLIC
    CONAN_PKG::fmt
    #append more dependencies here
)
